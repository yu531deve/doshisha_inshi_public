\section*{第1問（50点）}

図の一自由度振動系に関する (1)〜(4) に答えよ. なお, 質量を $m$, 粘性係数を $d$,
バネ定数を $k$ として, $f(t)$ は系に加える力, $x(t)$ は質量中心を表す.

\begin{center}
\begin{tikzpicture}[>=Latex]
  % 壁
  \draw[thick] (0,-0.2) -- (0,2.2);
  \foreach \y in {-0.1,0.1,...,2.05} \draw[thick] (0,\y) -- (-0.25,\y-0.25);
  % バネ k（上）
  \draw[thick,decorate,decoration={zigzag,segment length=8,amplitude=5}] (0,1.6) -- (3,1.6);
  \node at (1.5,2.05) {$k$};
  % ダンパ d（下）
  \draw[thick] (0,0.45) -- (1.15,0.45);
  \draw[thick] (1.15,0.2) rectangle (1.85,0.7);
  \draw[thick] (1.55,0.45) -- (3,0.45);
  \node at (1.5,-0.05) {$d$};
  % 質量 m
  \draw[thick,fill=gray!12] (3,0.1) rectangle (3.9,2.0);
  \node at (3.45,1.05) {$m$};
  % 力 f(t)
  \draw[->,thick] (3.9,1.05) -- (5.0,1.05);
  \node at (5.25,1.05) {$f(t)$};
  % 変位 x(t)
  \draw[->,thick] (3.45,2.35) -- (4.3,2.35);
  \node at (3.9,2.65) {$x(t)$};
\end{tikzpicture}
\end{center}

\begin{enumerate}[label=(\arabic*)]
  \item 入力を $f(t)$, 出力を $x(t)$ として, システムの伝達関数を求めよ.

  \item 物理パラメータを $m=1$, $d=1$ とし, 初期値を $x(0)=0$, $\dot{x}(0)=0$ とする.
        入力 $f(t)$ を単位ステップ関数とする. このとき $x(t)$ の定常値が $2$ になるように
        $k$ を定めよ.

  \begin{rem}
  原本（手書き）では本小問のパラメータ指定および「$\square$ を定めよ」の対象記号が判読困難である.
  単位ステップ入力に対する定常値 $1/k$ は粘性係数 $d$ に依存しないため,
  定めるべきパラメータは $k$ と解釈した（$m=1,\ d=1$ のとき定常値 $1/k=2 \Rightarrow k=1/2$）。
  \end{rem}

  \item 物理パラメータを $m=1$, $d=0$, $k=1$, 初期値を $x(0)=0$, $\dot{x}(0)=0$ とし,
        入力 $f(t)$ を単位ステップ関数とする. このとき $x(t)$ を求めよ. 加えて,
        $x(t)$ の概形を示せ. なお, 横軸を時間に, 縦軸を $x(t)$ とすること.

  \item 物理パラメータを $m=1$, $d=1$, $k=1$, 初期値を $x(0)=1$, $\dot{x}(0)=0$ とし,
        入力を $f(t)=0$ とする. このとき $x(t)$ を求めよ. また, $x(t)$ の定常値が
        存在すればその値も求めよ.
\end{enumerate}
